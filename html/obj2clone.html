Array to Clone
var ar = [];
		
ar.push('abc'); //String
ar.push(1.23);	//Number
ar.push(true);	//Boolean
ar.push({name:'Alex', fn:function(num){return num*2} }); //Object
ar.push(['xyz', 7.89, false]); //Array
ar.push(new Date());	//Date
ar.push(function(num){return num+1});	//Function
ar.push(/(\w+)\s(\w+)/); //RegExp
