Parse

var obj = JSONfn.parse(jsonfnObj);

// set the second arg true if jsonfnObj
// has property instance of Date
var obj = JSONfn.parse(jsonfnObj, true);
Examples: 


function showFirstName(){
  var str = JSONfn.stringify(jsonfnObj),
      objfn = JSONfn.parse(str);
  alert(objfn.firstName);
}



function showFullName(){
  var str = JSONfn.stringify(jsonfnObj),
      objfn = JSONfn.parse(str);
  alert(objfn.getFullName());
}



function showFullNameArrow(){
  var str = JSONfn.stringify(jsonfnObj),
      objfn = JSONfn.parse(str);
  alert(objfn.getFullNameArrow());
}



/* ES6 Arrow function inside Lambda */
function showLambdaGreeting(){
  var str = JSONfn.stringify(jsonfnObj),
      objfn = JSONfn.parse(str);
  alert(objfn.greetLambda('Hello '));
}



function testRegexp(){
  var str = JSONfn.stringify(jsonfnObj),
      objfn = JSONfn.parse(str),
      name = objfn.getFullName();
  alert(name.replace(objfn.re, "$2, $1"));
}



function testDate(){
  // pay attention at the 2-nd argument. Set it true if
  // original object had property instance of Date.
  var str = JSONfn.stringify(jsonfnObj),
      objfn = JSONfn.parse(str, true);
  alert(objfn.today instanceof Date);
}
